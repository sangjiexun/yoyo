6   package yoyo.actor.screen.iface.web.rest.home;
7   import java.math.BigDecimal;
8   import java.util.Collection;
9   import javax.ejb.EJB;
10  import javax.enterprise.context.RequestScoped;
11  import javax.ws.rs.Consumes;
12  import javax.ws.rs.DELETE;
13  import javax.ws.rs.GET;
14  import javax.ws.rs.POST;
15  import javax.ws.rs.Path;
16  import javax.ws.rs.PathParam;
17  import javax.ws.rs.Produces;
18  import javax.ws.rs.core.MediaType;
19  import yoyo.actor.screen.iface.web.rest.ResourceException;
20  import yoyo.actor.screen.iface.web.rest.ResourceFault;
21  import yoyo.actor.service.app.misc.NewsApplication;
22  import yoyo.actor.service.domain.misc.News;
23  import yoyo.framework.enterprise.domain.ConstraintsException;
24  /**
25   * News resources.
26   * @author nilcy
27   */
28  @Path("/home/news")
29  @RequestScoped
30  public class NewsResource {
31      /** news service Local-I/F. */
32      @EJB
33      private NewsApplication newsService;
34      /** ロガー */
35      // private final Logger log = Logger.getGlobal();
36      /** コンストラクタ */
37      public NewsResource() {
38          super();
39      }
40      /**
41       * news list.
42       * @return news list
43       */
44      @GET
45      @Path("/")
46      @Produces({ MediaType.TEXT_XML, MediaType.APPLICATION_JSON })
47      public Collection<News> getList() {
48          return newsService.findMany(null);
49      }
50      /**
51       * news.
52       * @param aId ID
53       * @return news
54       */
55      @GET
56      @Path("/{id}")
57      @Produces({ MediaType.TEXT_XML, MediaType.APPLICATION_JSON })
58      public News getItem(@PathParam("id") final BigDecimal aId) {
59          final News item = newsService.find(aId);
60          if (item == null) {
61              throw new ResourceException(new ResourceFault("ERR001", "GET_ERROR"));
62          }
63          return item;
64      }
65      /**
66       * Post news.
67       * @param aNews news
68       */
69      @POST
70      @Path("/")
71      @Consumes({ MediaType.TEXT_XML, MediaType.APPLICATION_JSON })
72      public void postItem(final News aNews) {
73          try {
74              newsService.save(aNews);
75          } catch (final ConstraintsException e) {
76              throw new ResourceException(new ResourceFault("ERR002", "POST_ERROR"));
77          }
78      }
79      /**
80       * Delete news.
81       * @param aId ID
82       */
83      @DELETE
84      @Path("/{id}")
85      @Consumes({ MediaType.TEXT_XML, MediaType.APPLICATION_JSON })
86      public void deleteItem(@PathParam("id") final BigDecimal aId) {
87          try {
88              newsService.delete(newsService.find(aId));
89          } catch (final ConstraintsException e) {
90              throw new ResourceException(new ResourceFault("ERR003", "DELETE_ERROR"));
91          }
92      }
93  }
