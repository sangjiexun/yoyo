13  public class MemberAuthFactory extends AbstractFactory<MemberAuth> {

18      /**
19       * Create instance.
20       * @see AbstractFactory#create()

22       * @param aAccount account
23       * @param aAuth authority

25       */
26      public MemberAuth create(final Member aMember, final String aAccount, final Auth aAuth) {

32      }
