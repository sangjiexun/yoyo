6   package yoyo.framework.enterprise.shared;
7   import java.text.MessageFormat;
8   import java.util.Locale;
9   import java.util.ResourceBundle;
10  /**
11   * Faces Helper.
12   * @author nilcy
13   */
14  public final class EnterpriseBundle {
15      /** コンストラクタ */
16      private EnterpriseBundle() {
17      }
18      /**
19       * bundle.
20       * @return bundle
21       */
22      private static ResourceBundle getBundle() {
23          return ResourceBundle.getBundle(EnterpriseProfile.BUNDLE_BASENAME, Locale.getDefault());
24      }
25      /**
26       * bundle message.
27       * @param aKey key
28       * @return bundle message
29       */
30      public static String getMessage(final String aKey) {
31          return getBundle().getString(aKey);
32      }
33      /**
34       * Format bundle message.
35       * @param aKey key
36       * @param aValues values
37       * @return bundle message
38       */
39      public static String formatMessage(final String aKey, final Object... aValues) {
40          return MessageFormat.format(getMessage(aKey), aValues);
41      }
42  }
