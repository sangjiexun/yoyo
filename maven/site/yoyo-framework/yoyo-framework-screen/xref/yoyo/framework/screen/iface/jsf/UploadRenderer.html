25  /**
26   * アップロードのレンダラ
27   * @author nilcy
28   */
29  @FacesRenderer(componentFamily = "javax.faces.Input", rendererType = "yoyo.faces.Upload")
30  public class UploadRenderer extends Renderer {
31      /** inputタグ */
32      private static final String TAG_INPUT = "input";
33      /** typeタグ */
34      private static final String TAG_TYPE = "type";
35      /** fileタグ */
36      private static final String TAG_FILE = "file";
37      /** nameタグ */
38      private static final String TAG_NAME = "name";
39      /** clientIdタグ */
40      private static final String TAG_CLIENT_ID = "clientId";
41      /** 記録者 */
42      private static final Logger LOG = Logger.getAnonymousLogger();
43      /** 実体化 */
44      public UploadRenderer() {
45          super();
46      }
47      /** {@inheritDoc} */
48      @SuppressWarnings("resource")
49      @Override
50      public void encodeBegin(final FacesContext context, final UIComponent component)
51          throws IOException {
52          if (!component.isRendered()) {
53              return;
54          }
55          final ResponseWriter writer = context.getResponseWriter();
56          final String clientId = component.getClientId(context);
57          writer.startElement(TAG_INPUT, component);
58          writer.writeAttribute(TAG_TYPE, TAG_FILE, TAG_TYPE);
59          writer.writeAttribute(TAG_NAME, clientId, TAG_CLIENT_ID);
60          writer.endElement(TAG_INPUT);
61          writer.flush();
62      }
63      /** {@inheritDoc} */
64      @Override
65      public void decode(final FacesContext context, final UIComponent component) {
66          final ExternalContext external = context.getExternalContext();
67          final HttpServletRequest request = (HttpServletRequest) external.getRequest();
68          final String clientId = component.getClientId(context);
69          final FileItem item = (FileItem) request.getAttribute(clientId);
70          if ((item == null) || StringUtils.isEmpty(item.getName())) {
71              return;
72          }
73          LOG.fine("item:" + item);
74          final ValueExpression valueExpr = component.getValueExpression("value");
75          if (valueExpr != null) {
76              ((EditableValueHolder) component).setSubmittedValue(findSubmittedValue(request, item,
77                  valueExpr.getType(context.getELContext())));
78              ((EditableValueHolder) component).setValid(true);
79          }
80          final Object target = component.getAttributes().get("target");
81          LOG.fine("target=" + target);
82          if (target != null) {
83              File file;
84              if (target instanceof File) {
85                  file = (File) target;
86              } else {
87                  final ServletContext servletContext = (ServletContext) external.getContext();
88                  final String itemName = item.getName();
89                  final String dirPath = "WEB-INF/" + target;
90                  LOG.fine("dirPath=" + dirPath);
91                  new File(servletContext.getRealPath(dirPath)).mkdirs();
92                  final String filePath = dirPath + "/" + itemName;
93                  final String realPath = servletContext.getRealPath(filePath);
94                  LOG.fine("realPath=" + realPath);
95                  file = new File(realPath);
96              }
97              // UGH 汎用例外クラスを捕捉したくないがcommons-upload仕様が…
98              try {
99                  LOG.fine("item.write");
100                 item.write(file);
101             } catch (final Exception ex) {
102                 throw new FacesException(ex);
103             }
104         }
105     }
106     /**
107      * 画面値の取得
108      * @param request HTTP要求
109      * @param item FILE項目
110      * @param valueType 値型
111      * @return 画面値
112      * @throws FacesException 画面例外
113      */
114     @SuppressWarnings("static-method")
115     private Object findSubmittedValue(final HttpServletRequest request, final FileItem item,
116         final Class<?> valueType) throws FacesException {
117         Object newValue;
118         if (valueType == byte[].class) {
119             newValue = item.get();
120         } else if (valueType == InputStream.class) {
121             try {
122                 newValue = item.getInputStream();
123             } catch (final IOException ex) {
124                 throw new FacesException(ex);
125             }
126         } else {
127             final String encoding = request.getCharacterEncoding();
128             if (encoding != null) {
129                 try {
130                     newValue = item.getString(encoding);
131                 } catch (final UnsupportedEncodingException ex) {
132                     newValue = item.getString();
133                 }
134             } else {
135                 newValue = item.getString();
136             }
137         }
138         return newValue;
139     }
