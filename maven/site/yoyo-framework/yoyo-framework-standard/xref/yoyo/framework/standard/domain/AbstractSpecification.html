6   package yoyo.framework.standard.domain;
7   /**
8    * 仕様オブジェクト
9    * <p>
10   * 論理演算をするための値オブジェクトである。ビジネスルールを「述語」的に表現するときに使用する。
11   * </p>
12   * @param <T> チェック対象オブジェクト型
13   * @author nilcy
14   */
15  public abstract class AbstractSpecification<T> implements Specification<T> {
16      /** {@inheritDoc} */
17      @Override
18      public abstract boolean isSatisfiedBy(T object);
19      /** {@inheritDoc} */
20      @Override
21      public Specification<T> and(final Specification<T> spec) {
22          return new AndSpecification<>(this, spec);
23      }
24      /** {@inheritDoc} */
25      @Override
26      public Specification<T> or(final Specification<T> spec) {
27          return new OrSpecification<>(this, spec);
28      }
29      /** {@inheritDoc} */
30      @Override
31      public Specification<T> not(final Specification<T> spec) {
32          return new NotSpecification<>(spec);
33      }
34      /** {@inheritDoc} */
35      @Override
36      public Specification<T> xor(final Specification<T> aSpecification) {
37          return new XorSpecification<>(this, aSpecification);
38      }
39      /** {@inheritDoc} */
40      @Override
41      public Specification<T> nand(final Specification<T> aSpecification) {
42          return new NandSpecification<>(this, aSpecification);
43      }
44      /** {@inheritDoc} */
45      @Override
46      public Specification<T> nor(final Specification<T> aSpecification) {
47          return new NorSpecification<>(this, aSpecification);
48      }
49  }
