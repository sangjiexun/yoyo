6   package yoyo.actor.screen.iface.jsf.application;
7   import java.io.BufferedReader;
8   import java.io.IOException;
9   import java.io.InputStream;
10  import java.io.InputStreamReader;
11  import java.io.Serializable;
12  import java.io.UnsupportedEncodingException;
13  import java.util.HashMap;
14  import java.util.Map;
15  import javax.annotation.PostConstruct;
16  import javax.ejb.Singleton;
17  import javax.enterprise.context.ApplicationScoped;
18  import javax.inject.Named;
19  import org.apache.commons.lang3.Validate;
20  import yoyo.actor.service.domain.parts.Address;
21  import yoyo.actor.service.domain.parts.HomeAddress;
22  import yoyo.actor.service.domain.parts.HomeAddressFactory;
23  import yoyo.framework.enterprise.shared.EnterpriseRuntimeException;
24  /**
25   * Home address action.
26   * @author nilcy
27   */
28  @Named("homeAddressAction")
29  @ApplicationScoped
30  @Singleton
31  // @Startup
32  public class HomeAddressAction implements Serializable {
33      /** 製品番号 */
34      private static final long serialVersionUID = -4847634579319468521L;
35      /** LEN_CODE. */
36      private static final int LEN_CODE = 7;
37      /** POS_CODE. */
38      private static final int POS_CODE = 2;
39      /** POS_PREF. */
40      private static final int POS_PREF = 6;
41      /** POS_TOWN. */
42      private static final int POS_TOWN = 7;
43      /** POS_CITY. */
44      private static final int POS_CITY = 8;
45      /** address map. */
46      private Map<String, HomeAddress> map;
47      /** コンストラクタ */
48      public HomeAddressAction() {
49          super();
50      }
51      /**
52       * Process on post-construct.
53       */
54      @PostConstruct
55      public void postConstruct() {
56          try {
57              try (InputStream is = this.getClass().getResourceAsStream("/address/HomeAddress.csv");
58                  BufferedReader br = new BufferedReader(new InputStreamReader(is, "Windows-31J"));) {
59                  String line;
60                  map = new HashMap<>();
61                  while ((line = br.readLine()) != null) {
62                      final String[] tmp = line.split(",");
63                      final String code = trimQuote(tmp[POS_CODE]);
64                      final String pref = trimQuote(tmp[POS_PREF]);
65                      final String town = trimQuote(tmp[POS_TOWN]);
66                      final String city = trimQuote(tmp[POS_CITY]);
67                      final HomeAddress address = new HomeAddressFactory().create(code, pref, town,
68                          city);
69                      map.put(code, address);
70                  }
71              }
72          } catch (final UnsupportedEncodingException e) {
73              throw new EnterpriseRuntimeException(e);
74          } catch (final IOException e) {
75              throw new EnterpriseRuntimeException(e);
76          }
77      }
78      /**
79       * Trim item.
80       * @param aItem item
81       * @return trimmed item
82       */
83      private static String trimQuote(final String aItem) {
84          return aItem.replaceAll("\"", "");
85      }
86      /**
87       * Load home address.
88       * @param aAddress home address
89       * @return true if found
90       */
91      public boolean load(final Address<?> aAddress) {
92          Validate.notNull(aAddress);
93          final String key = aAddress.getCode().replaceAll("-", "");
94          if (key.length() != LEN_CODE) {
95              return false;
96          }
97          final HomeAddress address = map.get(key);
98          if (address != null) {
99              aAddress.setPref(address.getPref());
100             aAddress.setCity(address.getCity());
101             aAddress.setTown(address.getTown());
102         }
103         return address != null;
104     }
