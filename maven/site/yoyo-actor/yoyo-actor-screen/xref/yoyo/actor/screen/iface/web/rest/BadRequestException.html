6   package yoyo.actor.screen.iface.web.rest;
7   import java.util.Arrays;
8   import java.util.List;
9   import javax.ws.rs.WebApplicationException;
10  import javax.ws.rs.core.GenericEntity;
11  import javax.ws.rs.core.MediaType;
12  import javax.ws.rs.core.Response;
13  import javax.ws.rs.core.Response.Status;
14  /**
15   * Thrown to return a 400 Bad Request response with a list of error messages in
16   * the body.
17   * @author nilcy
18   */
19  public class BadRequestException extends WebApplicationException {
20      /** 製品番号 */
21      private static final long serialVersionUID = -3044929615860691176L;
22      /** Errors. */
23      private final List<String> errors;
24      /**
25       * コンストラクタ
26       * @param aErrors errors
27       */
28      public BadRequestException(final String... aErrors) {
29          this(Arrays.asList(aErrors));
30      }
31      /**
32       * コンストラクタ
33       * @param aErrors errors
34       */
35      public BadRequestException(final List<String> aErrors) {
36          super(Response.status(Status.BAD_REQUEST).type(MediaType.APPLICATION_XHTML_XML)
37              .entity(new GenericEntity<List<String>>(aErrors) {
38              }).build());
39          this.errors = aErrors;
40      }
41      /**
42       * errors.
43       * @return errors
44       */
45      public List<String> getErrors() {
46          return this.errors;
47      }
48  }
