6   package yoyo.framework.standard.shared;
7   import static org.junit.Assert.*;
8   import java.text.MessageFormat;
9   import java.util.logging.Logger;
10  import org.apache.commons.codec.binary.Hex;
11  import org.junit.Test;
12  import yoyo.framework.standard.shared.SecurityUtils;
13  /**
14   * {@link SecurityUtils} のユニットテスト
15   * @author nilcy
16   */
17  @SuppressWarnings("all")
18  public class SecurityUtilsTest {
19      private static final Logger LOG = Logger.getAnonymousLogger();
20      @Test
21      public void testCreateKey() {
22          final String key01 = SecurityUtils.createKey();
23          final String key02 = SecurityUtils.createKey();
24          assertTrue(key01 != key02);
25          LOG.info(MessageFormat.format("共通鍵01={0},共通鍵02={1}", key01, key02));
26      }
27      @Test
28      public void testEncryptDecrypt() {
29          final String key = SecurityUtils.createKey();
30          final String plainText = "極秘指令書その壱だお";
31          final byte[] rawBytes = plainText.getBytes();
32          final byte[] encBytes = SecurityUtils.encrypt(rawBytes, key);
33          final byte[] decBytes = SecurityUtils.decrypt(encBytes, key);
34          LOG.info(MessageFormat.format(
35              "raw hex={0},encrypted hex={1},decrypted string={2}",
36              new Object[] { Hex.encodeHexString(rawBytes), Hex.encodeHexString(encBytes),
37                  Hex.encodeHexString(decBytes) }));
38          assertTrue(Hex.encodeHexString(rawBytes) != Hex.encodeHexString(encBytes));
39          assertTrue(Hex.encodeHexString(encBytes) != Hex.encodeHexString(decBytes));
40          assertEquals(Hex.encodeHexString(rawBytes), Hex.encodeHexString(decBytes));
41          LOG.info(MessageFormat.format("raw string={0},decrypted string={1}", new String(rawBytes),
42              new String(decBytes)));
43          assertEquals(new String(rawBytes), new String(decBytes));
44      }
45  }
