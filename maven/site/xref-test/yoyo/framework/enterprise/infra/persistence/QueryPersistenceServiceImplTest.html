32  public class QueryPersistenceServiceImplTest {
33      @Inject
34      @TestExamplePersistenceService
35      private QueryPersistenceService<TestExample, BigDecimal> testee;
36      @Inject
37      private Logger log;
38      @Test
39      public void testCRUD() {
40          final TestExample e01 = new TestExample("code01");
41          testee.persist(e01);
42          assertThat(e01.getId(), is(BigDecimal.valueOf(1L)));
43          assertThat(testee.contains(e01), is(true));
44          final TestExample e01r = testee.find(BigDecimal.valueOf(1L));
45          assertThat(e01r, is(e01));
46          e01r.setCode("code01#1");
47          testee.lock(e01r, LockModeType.PESSIMISTIC_READ);
48          testee.flush();
49          testee.refresh(e01r);
50          testee.refresh(e01r, LockModeType.PESSIMISTIC_WRITE);
51          testee.detach(e01r);
52          e01r.setCode("code01#2");
53          testee.merge(e01r);
54          testee.remove(testee.find(e01r.getId()));
55          assertThat(testee.find(BigDecimal.valueOf(1L), LockModeType.OPTIMISTIC), is(nullValue()));
56      }
57      @Test
58      public void testCriteria() {
59          final CriteriaBuilder b = testee.builder();
60          final Root<TestExample> r = testee.root();
61          testee.persist(new TestExample("code01"));
62          testee.persist(new TestExample("code02"));
63          testee.query().select(r).where(b.equal(r.get(code), "code01")).orderBy(b.asc(r.get(code)));
64          final Collection<TestExample> testExamples = testee.createQuery().getResultList();
65          assertThat(testExamples.size(), is(1));
66          for (final TestExample e : testExamples) {
67              assertThat(e.getCode(), is("code01"));
68              log.info("TestExampleObject >> " + e.toString());
69          }
70      }
